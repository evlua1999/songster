\documentclass{standalone}

\usepackage[l3]{csvsimple}
\usepackage{qrcode}

\pdfpagewidth=210 true mm
\pdfpageheight=297 true mm

\newdimen\rulesize
\rulesize=0.2pt

\newbox\dummy
\output={\setbox\dummy=\box255\deadcycles=0}

\newbox\frontbox
\newbox\backbox

\def\mkcard{
  \setbox\frontbox\hbox{{\hsize=6.5cm\relax\vbox to 6.5cm{\vss\noindent\hskip0.5cm\parbox{5.5cm}{\noindent\begin{center}\qrcode[height=5.5cm]{\link}\end{center}}\hskip0.5cm\vss}}}
  \setbox\backbox\hbox{{\hsize=6.5cm\relax\vbox to 6.5cm{\vss\noindent\hskip0.5cm\parbox{5.5cm}{\noindent\begin{center}{\Large\name}\vskip5pt{\large\artist}\vskip5pt{\large\year}\end{center}}\hskip0.5cm\vss}}}
  \dp\frontbox=0pt\dp\backbox=0pt\ht\frontbox=6.5cm\ht\backbox=6.5cm\wd\frontbox=6.5cm\wd\backbox=6.5cm\relax
}

\newbox\frontrow
\newbox\backrow

\newbox\frontpage
\newbox\backpage
\newbox\sailor

\newcount\colcount
\newcount\rowcount
\colcount=0\rowcount=0

\def\flushbox{
  \setbox\frontrow\hbox to \pdfpagewidth{\unhbox\frontrow\vrule width \rulesize height 6.5cm\relax\box\frontbox}
  \setbox\backrow\hbox to \pdfpagewidth{\box\backbox\vrule width \rulesize height 6.5cm\relax\unhbox\backrow}
  \advance\colcount by 1
  \ifnum\colcount=3\relax\flushrow\fi
}

\def\flushrow{
  \setbox\frontrow\hbox to \pdfpagewidth{\hss\unhbox\frontrow\vrule width \rulesize height 6.5cm\relax\hss}
  \setbox\frontpage\vbox{\unvbox\frontpage\hrule height \rulesize width \pdfpagewidth\relax\box\frontrow}
  \setbox\backrow\hbox to \pdfpagewidth{\hss\vrule width \rulesize height 6.5cm\relax\unhbox\backrow\hss}
  \setbox\backpage\vbox{\unvbox\backpage\hrule height \rulesize width \pdfpagewidth\relax\box\backrow}
  \colcount=0
  \advance\rowcount by 1
  \ifnum\rowcount=4\relax\flushpage\fi
}

\def\flushpage{
  \setbox\frontpage\vbox{\unvbox\frontpage\hrule height \rulesize width \pdfpagewidth\relax}
  \setbox\sailor\vbox to \pdfpageheight{\vss\box\frontpage\vss}
  \shipout\box\sailor
  \setbox\backpage\vbox{\unvbox\backpage\hrule height \rulesize width \pdfpagewidth\relax}
  \setbox\sailor\vbox to \pdfpageheight{\vss\box\backpage\vss}
  \shipout\box\sailor
  \rowcount=0
}


\begin{document}
  \hsize=\pdfpagewidth
  \csvreader[head to column names]{out.csv}{}{\mkcard\flushbox}
\end{document}
